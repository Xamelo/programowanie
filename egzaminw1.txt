using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace czerwiec2025aplikacja
{
    internal class MainPageViewModel : BindableObject
    {
        private int colorRvalue;
        public int ColorRvalue
        {
            get { return colorRvalue; }
            set { colorRvalue = value; OnPropertyChanged(); }
        }

        private int colorGvalue;
        public int ColorGvalue
        {
            get { return colorGvalue; }
            set { colorGvalue = value; OnPropertyChanged(); }
        }

        private int colorBvalue;
        public int ColorBvalue
        {
            get { return colorBvalue; }
            set { colorBvalue = value; OnPropertyChanged(); }
        }

        public Color LabelColor { get; set; }

        private string labelText;

        public string LabelText
        {
            get { return labelText; }
            set { labelText = ColorRvalue.ToString() + ", " + ColorGvalue.ToString() + ", " + colorBvalue.ToString(); OnPropertyChanged(); }
        }


        public MainPageViewModel()
        {
            LabelColor = Color.FromRgb(255, 255, 255);
            ColorRvalue = 255;
            ColorGvalue = 255;
            colorBvalue = 255;
            LabelText = ColorRvalue.ToString() + ", " + ColorGvalue.ToString() + ", " + colorBvalue.ToString();
        }
    }
}
